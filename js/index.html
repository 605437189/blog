# 深度优先遍历算法

  深度优先遍历DFS 与树的先序遍历比较类似。假设初始状态是图中所有顶点均未被访问，则从某个顶点v出发，首先访问该顶点然后依次从它的各个未被访问的邻接点出发深度优先搜索遍历图，直至图中所有和v有路径相通的顶点都被访问到。若此时尚有其他顶点未被访问到，则另选一个未被访问的顶点作起始点，重复上述过程，直至图中所有顶点都被访问到为止。

  //遍历BODY下面的dom节点
  function dfs(node, nodeList = []){
    if(node !== null){
      nodeList.push(node)
    }
    let children = node.children
    for(let i = 0 ; i < children.length; i++){
      dfs(children[i], nodeList)
    }
    return nodeList
  }
  dfs(document.body)

  function dfs1(node){
    var nodeList = []
    if(node !== null){
      nodeList.push(node)
    }
    const children = node.children
    for(let i = 0; i< children.length; i++){
      nodeList = nodeList.concat(dfs1(children[i]))
    }
    return nodeList
  }

  function dfs2(node){
    let stack = []
    let nodeList = []
    if(node){
      stack.push(node)
      while(stack.length){
        let item = stack.pop()
        node.push(item)
        let children = node.children
        for(let i =children.length -1 ; i>= 0; i --){ //这里是从后往前遍历哦
          stack.push(children[i])
        }
      }
    }
    return nodeList
  }
